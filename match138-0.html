def swap_child(root, level, M):


    if root is None:
        return root


    if level < M:
        tmp0 = root.left
        root.left = root.right
        root.right = tmp0


    swap_child(root.left,level+1, M)


    swap_child(root.right,level+1, M)
    return root

def subtract_summation(root):
 
    def extSum(t):
        if t is None:
            return 0
        return t.elem + extSum(t.left) + extSum(t.right)


    if root is None:
        return 0
    leftSum  = extSum(root.left)
    rightSum = extSum(root.right)
    ans = leftSum - rightSum


    return ans

def level_sum(root):
    def go(now,lvl):
        if now==None:
            return 0
        if lvl%2==0:
            here = -now.elem
        else:
            here = now.elem
        return here + go(now.left,lvl+1) + go(now.right,lvl+1)
    return go(root,0)

def rangeSum(root, low, high):
    if root is None:
        return 0


    val = root.elem


    if val < low:
        return rangeSum(root.right, low, high)


    if val > high:
        return rangeSum(root.left, low, high)


    lefty  = rangeSum(root.left,  low, high)
    righty = rangeSum(root.right, low, high)
   
    ans = val + lefty + righty
    return ans

def mirror_sum(root):
    if root is None:
        return 0


    def smoosh(a,b):
        if a is None or b is None:
            return 0
        this_pair = a.elem + b.elem
        side1 = smoosh(a.left,  b.right)
        side2 = smoosh(a.right, b.left)
        return this_pair + side1 + side2


    return smoosh(root.left, root.right)

def isBST(root):
    def walk(t, low, high):
        if t is None:
            return True
        x = t.elem
        if low is not None and x <= low:
            return False
        if high is not None and x >= high:
            return False
        left_ok  = walk(t.left,low, x)
        right_ok = walk(t.right, x,high)
        return left_ok and right_ok


    return walk(root, None, None)
