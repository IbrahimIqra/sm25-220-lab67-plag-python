def swap_child(root, level, M):
    if root == None:
        return None
    if (level+1)<=M:
      root.left,root.right=root.right,root.left
    swap_child(root.left,level+1,M)
    swap_child(root.right,level+1,M)
    return root

def level_sum(root):
  def helper(root,level=1):
    if root==None:
      return 0


    if level%2==1:
      return -root.elem+helper(root.left,level+1)+helper(root.right,level+1)
    else:
      return root.elem+helper(root.left,level+1)+helper(root.right,level+1)
  return helper(root, 1)

def rangeSum(root, low, high):
  if root==None:
    return 0
  if root.elem> high:
    temp1=rangeSum(root.left, low, high)
    return temp1
  if root.elem <low:
    temp2=rangeSum(root.right, low, high)
    return temp2
    
  return root.elem +rangeSum(root.left, low, high) +rangeSum(root.right, low, high)

def mirror_sum(root):
  if root==None:
    return 0
  def helper(left,right):
    if left==None or right==None:
      return 0
    sum=left.elem +right.elem
    return sum+ helper(left.left,right.right)+helper(left.right,right.left)
  return helper(root.left,root.right)

    def helper(root, min, max):
        if root is None:
            return False


        if not (min < root.elem < max):
            return False
        return (helper(root.left, min, root.elem) and helper(root.right, root.elem, max))
