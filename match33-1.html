def subtract_summation(root):
  if root == None:
    return 0
  if root.left == None and root.right == None:
    return root.elem
  left_val = subtract_summation(root.left)
  right_val = subtract_summation(root.right)
  return root.elem - (left_val + right_val)

def level_sum(root):
  def helper(node, level):


    if node == None:
      return 0, 0  
    left_odd, left_even = helper(node.left, level+1)
    right_odd, right_even = helper(node.right, level+1)
    if level % 2 == 1:
      X = node.elem + left_odd + right_odd, left_even + right_even
      return X
    else:
      Y = left_odd + right_odd, node.elem + left_even + right_even            
      return Y
  odd, even = helper(root, 1)


  return abs(odd - even)

def rangeSum(root, low, high):
  total = 0
  if root == None:
    return 0


  if root.elem >= low and root.elem <= high:
    total = total + root.elem
  if root.elem > low:
    total = total + rangeSum(root.left, low, high)
  if root.elem < high:
    total = total + rangeSum(root.right, low, high)
  return total

def mirror_sum(root):
  def helper(left_node, right_node):
    if left_node == None or right_node == None:


      return 0
    total = left_node.elem + right_node.elem
    total = total + helper(left_node.left, right_node.right)
    total = total + helper(left_node.right, right_node.left)
    return total
  if root == None:
    return 0
  return helper(root.left, root.right)

def isBST(root):
  def helper(node, min_value, max_value):
    if node == None:


      return True
    if node.elem <= min_value or node.elem >= max_value:
      return False
    left_ok = helper(node.left, min_value, node.elem)
    right_ok = helper(node.right, node.elem, max_value)
    if left_ok == False or right_ok == False:
      return False
    return True
  return helper(root, float('-inf'), float('inf'))
