def swap_child(root, level, M):
    if root is None:
        return

    if level >= M:
        root.left, root.right = root.right, root.left

    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)

    return root

def level_sum(root):
  #To Do
  if root is None:
    return 0

  q = [(root, 0)]
  even_sum = 0
  odd_sum = 0

  while q:
    node, level = q.pop(0)

    if level % 2 == 0:
      even_sum += node.elem
    else:
      odd_sum += node.elem

    if node.left:
      q.append((node.left, level + 1))
    if node.right:
      q.append((node.right, level + 1))



  return odd_sum - even_sum

def rangeSum(root, low, high):
  if root is None:
    return 0
  total_sum = 0
  if root.elem >= low and root.elem <= high:
    total_sum += root.elem
  if root.elem > low:
    total_sum += rangeSum(root.left, low, high)
  if root.elem < high:
    total_sum += range Sum(root.right, low, high)
  return total_sum

def mirror_sum_helper(left_node, right_node):
  if left_node is None or right_node is None:
    return 0
  current_sum = left_node.elem + right_node.elem
  current_sum += mirror_sum_helper(left_node.left, right_node.right)
  current_sum += mirror_sum_helper(left_node.right, right_node.left)
  return current_sum


def mirror_sum(root):
  if root is None:
    return 0
  if root.left is None or root.right is None:
      return 0
  return mirror_sum_helper(root.left, root.right)

def isBST_helper(root, min_val, max_val):
  if root is None:
    return True
  if root.elem <= min_val or root.elem >= max_val:
    return False
  return (isBST_helper(root.left, min_val, root.elem) and
          isBST_helper(root.right, root.elem, max_val))


def isBST(root):
  return isBST_helper(root, float('-inf'), float('inf'))
