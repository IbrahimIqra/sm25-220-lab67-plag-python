def level_sum(root):
  def helper(node,level):
   if node is None:
     return 0
   if level % 2 != 0:
     return (node.elem + helper(node.right, level+1) + helper(node.left,level+1))
   else:
     return (-node.elem + helper(node.right, level+1) + helper(node.left, level+1)) 
  return helper(root,0)

def rangeSum(root, low, high):
  if root is None:
    return 0


  if root.elem < low:
        return rangeSum(root.right, low, high)


  elif root.elem > high:
        return rangeSum(root.left, low, high)


  return (root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high))

def isBST(root):
 
    def helper(node,l,h):
        if node is None:
            return True
        if (l is not None and node.elem <=l) or (h is not None and node.elem >=h):
            return False
        return helper(node.left,l, node.elem) and helper(node.right, node.elem,h)


    return helper(root, None, None)
