def rangeSum(r,l,h):
    if r is None:
        return 0
    t=0
    if r.elem>=l and r.elem<=h:
        t=t+r.elem
    if r.elem>l:
        total=t+rangeSum(r.left,l,h)
    if r.elem<h:
        total=t+rangeSum(r.right,l,h)
    return total

def mirror_sum(r):
    def helper(left,right):
        if left is None or right is None:
            return 0
        s=left.elem+right.elem
        s=s+helper(left.left,right.right)
        s=s+helper(left.right,right.left)
        return s
    if r is None:
        return 0
    return helper(r.left,r.left)
