def swap_child(root, level, M):


  if root == None:
    return None
  if level >= M:
    store = root.left
    root.left = root.right
    root.right = store
  swap_child(root.left, level+1, M)
  swap_child(root.right, level+1, M)
  return root

def rangeSum(root, low, high):
  if root == None:
    return 0

  total = 0
  if root.elem > low:
    total += rangeSum(root.left, low, high)
  if root.elem < high:
    total += rangeSum(root.right, low, high)
  if root.elem >= low and root.elem <= high:
    total += root.elem
  return total

def mirror_sum(root):
  if root == None:
    return 0
  return root_sum(root.left, root.right)

def root_sum(left, right):
  if left == None or right == None:
    return 0
  return left.elem + right.elem + root_sum(left.left, right.right) + root_sum(left.right, right.left)
