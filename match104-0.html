def rangeSum(root, low, high):
    if root is None:
        return 0
    if root.elem < low:
        return rangeSum(root.right,low,high)
    if root.elem > high:
        return rangeSum(root.left,low,high)
    return root.elem+rangeSum(root.left,low,high)+rangeSum(root.right,low,high)

def mirror_sum(root):
    def helper_func(left,right):
        if left is None or right is None:
            return 0
        sum1=left.elem+right.elem
        sum1+=helper_func(left.left,right.right)
        sum1+=helper_func(left.right,right.left)
        return sum1
    if root is None:
        return 0
    return helper_func(root.left, root.right)

def isBST(root):
    def helper_func(node,minimum_value,maximum_value):
        if node is None:
            return True
        if not (minimum_value<node.elem<maximum_value):
            return False
        return helper_func(node.left,minimum_value,node.elem) and helper_func(node.right,node.elem,maximum_value)

    return helper_func(root,int(),int())
