def swap_child(root, level, M):
  


  if root is None:
    return None
  
  if level < M:
    temp = root.left
    root.left = root.right
    root.right = temp
  
  swap_child(root.left, level + 1, M)
  swap_child(root.right, level + 1, M)
  
  return root

def subtract_summation(root):
  
  def helper(node):
    if node is None:
      return 0
    return node.elem + helper(node.left) + helper(node.right)
  
  if root == None:
    return 0
  
  sum_left = helper(root.left)
  sum_right = helper(root.right)
  
  return sum_left - sum_right

def level_sum(root):
  def helper_fn(node, level):
    if node is None:
      return 0
    if level % 2 == 1:
      sign = 1
    else:
      sign = -1


    return sign * node.elem + helper_fn(node.left, level + 1) + helper_fn(node.right, level + 1)
  
  return helper_fn(root, 0)

def rangeSum(root, low, high):
    if root is None:
        return 0
    
    if root.elem > high:
        return rangeSum(root.left, low, high)
    


    if root.elem < low:
        return rangeSum(root.right, low, high)
    
    return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
    
    def helper_func(left, right):


        if left is None or right is None:
            return 0
        return left.elem + right.elem + helper_func(left.left, right.right) + helper_func(left.right, right.left)
    
    if root is None:
        return 0
    
    return helper_func(root.left, root.right)
