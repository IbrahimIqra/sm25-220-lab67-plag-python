def swap_child(root, level, M):
  if root==None:
    return None
  if level>=M:
    val=root.left
    root.left=root.right
    root.right=val
  swap_child(root.left,level+1,M)


  swap_child(root.right,level+1,M)
  return root

def subtract_summation(root):
  def func(node):
    if node==None:
      return 0
    l_sum=func(node.left)
    r_sum=func(node.right)
    return node.elem + l_sum + r_sum
  if root==None:
      return 0
  l_tot=func(root.left)
  r_tot=func(root.right)


  return root.elem-(l_tot+r_tot)

def rangeSum(root, low, high):
  if root==None:
    return 0
  tot=0


  if root.elem>low:
    tot+=rangeSum(root.left,low,high)
  if root.elem<high:
    tot+=rangeSum(root.right,low,high)
  if low<=root.elem<=high:
    tot+=root.elem
  return tot

def mirror_sum(root):
  def func(l,r):
    if l==None or r==None:
      return 0


    tot=l.elem+r.elem
    tot+=func(l.left,r.right)
    tot+=func(l.right,r.left)
    return tot
  if root==None:
    return 0
  return func(root.left,root.right)

def isBST(root):
  def func(node,min,max):
    if node==None:
      return True
    if node.elem<=min or node.elem>=max:
      return False
    return func(node.left,min,node.elem)
    return func(node.right,max,node.elem)
  return func(root,float("-inf"),float("inf"))
