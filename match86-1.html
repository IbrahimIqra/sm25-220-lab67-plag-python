    if root is None:
        return
    
    if level<M:
        temp= root.left
        root.left=root.right
        root.right=temp

        

    swap_child(root.left,level+1,M)
    swap_child(root.right,level+1,M)
    return root

def rangeSum(root, low, high):
    if root==None:
        return 0



    sum=0
    if low<=root.elem<=high:
        sum+=root.elem

    sum+=rangeSum(root.left, low, high)
    sum+=rangeSum(root.right, low, high)

    return sum

def mirror_sum(root):
    if root==None:
        return 0

    def helper(l,r):
        sum=0
        if l==None or r==None:
            return 0
            


        sum+=l.elem+r.elem
        sum+=helper(l.left, r.right)
        sum+=helper(l.right, r.left)

        return sum
        
    return helper(root.left, root.right)
