def swap_child(root,level,M):
    if root is None:
        return None
    if level<M:
        t=root.left
        root.left=root.right
        root.right=t
    swap_child(root.left,level+1,M)
    swap_child(root.right,level+1,M)
    return root

def level_sum(root):
  def f(node,level):
    if node is None:
      return 0
    s=f(node.left,level+1)+f(node.right,level+1)
    if level&1:
      s=node.elem+s
    else:
      s=-node.elem+s
    return s
  return f(root,0)

def rangeSum(root, low, high):

    if root is None:
        return 0
    v=0
    if root.elem>=low and root.elem<=high:
        v=root.elem


    if root.elem>low:
        v=v+rangeSum(root.left,low,high)
    if root.elem<high:
        v=v+rangeSum(root.right,low,high)
    return v

def isBST(root):


  def f(n,lo,hi):
    if n is None:
      return True
    if lo is not None and n.elem<=lo:
      return False
    if hi is not None and n.elem>=hi:
      return False
    if not f(n.left,lo,n.elem):
      return False
    if not f(n.right,n.elem,hi):
      return False
    return True
  return f(root,None,None)
