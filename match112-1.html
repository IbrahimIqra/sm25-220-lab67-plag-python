def rangeSum(root, low, high):


    if root is None:
       return 0
  
    if root.val < low:
       return rangeSum(root.right, low, high)
  
    if root.val > high:
       return rangeSum(root.left, low, high)
  
    return (root.val +
           rangeSum(root.left, low, high) +
           rangeSum(root.right, low, high))

def mirror_sum(root):
  def dfs(left, right):
      if not left or not right:
          return 0
    
      # Sum this pair
      pair_sum = left.val + right.val
    
      # Recursively sum for outer and inner children
      return pair_sum + dfs(left.left, right.right) + dfs(left.right, right.left)
   if not root:
      return 0
   return dfs(root.left, root.right)

def isBST(root):
   prev = [None] 


   def inorder(node):
       if node is None:
           return True
       if not inorder(node.left):
           return False
       if prev[0] is not None and node.val <= prev[0]:
           return False
       prev[0] = node.val
       return inorder(node.right)


   return inorder(root)
