def swap_child(root, level, M):
  if root is None:
    return root
  if level < M:


    tmp = root.left
    root.left = root.right
    root.right = tmp
    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)
  return root

        right_depth = find_depth(node.right)

def rangeSum(root, low, high):
    if root == None:
        return 0


    total = 0
    if root.elem >= low and root.elem <= high:
        total += root.elem
    if root.elem > low:
        total += rangeSum(root.left, low, high)
    if root.elem < high:
        total += rangeSum(root.right, low, high)


    return total

def mirror_sum(root):
    if root == None:
        return 0


    def checkMirror(a, b):
        if a == None or b == None:
            return 0




        s = a.elem + b.elem  
        s += checkMirror(a.left, b.right)
        s += checkMirror(a.right, b.left)


        return s


    return checkMirror(root.left, root.right)

def isBST(root):
    def check(node, low, high):
        if node == None:
            return True


        if node.elem <= low or node.elem >= high:
            return False


        left_ok = check(node.left, low, node.elem)
        right_ok = check(node.right, node.elem, high)


        return left_ok and right_ok


    return check(root, float('-inf'), float('inf'))
