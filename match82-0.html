def swap_child(root, level, M):
    if root:


        if level==M:
          root.right,root.left=root.left,root.right
        swap_child(root.left,level+1,M)
        swap_child(root.right,level+1,M)
        
    return root

def subtract_summation(root):
    if root is None:
        return 0
    


    def subtree_sum(root):
        if root is None:
            return 0
        return root.elem+subtree_sum(root.left)+subtree_sum(root.right)
    
    left_sum=subtree_sum(root.left)
    right_sum=subtree_sum(root.right)
    return left_sum-right_sum

def level_sum(root):
    if root is None:
        return 0
    left_depth=level_sum(root.left)
    right_depth=level_sum(root.right)
    return 1 +max(left_depth, right_depth)

def rangeSum(root, low, high):
    if root is None:
        return 0
    if root.elem<low:
        return rangeSum(root.right,low,high)
    if root.elem>high:
        return rangeSum(root.left,low,high)
    return root.elem+rangeSum(root.left,low,high)+rangeSum(root.right,low,high)

def mirror_sum(root):
    def helper(a,b):
        if a is None or b is None:
            return 0
        return a.elem+b.elem+helper(a.left,b.right)+helper(a.right,b.left)
    if root is None:
        return 0
    return helper(root.left,root.right)

def isBST(root):


    def check(node,low,high):
        if node is None:
            return True
        if node.elem<=low or node.elem>=high:
            return False
        return check(node.left,low,node.elem) and check(node.right,node.elem,high)
    return check(root,float('-inf'),float('inf'))
