        root.left=temp
        swap_child(root, level+1, M)

def level_sum(root):
    def helper(node, level):
        if node is None:
            return 0
        if level % 2 == 1:
            return node.data + helper(node.left, level + 1) + helper(node.right, level + 1)
        else:
            return -node.data + helper(node.left, level + 1) + helper(node.right, level + 1)
    return helper(root, 1)

def rangeSum(root, low, high):
    if root is None:
        return 0


    v = root.element   


    if v < low:
        return rangeSum(root.right, low, high)
    if v > high:
        return rangeSum(root.left, low, high)


    return v + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):


    def helper(left, right):
        if left is None or right is None:
            return 0
        return left.data + right.data + helper(left.left, right.right) + helper(left.right, right.left)


    if root is None:
        return 0
    return helper(root.left, root.right)

def isBST(root):
  def helper(node, level):
        if node is None:
            return 0
        if level % 2 == 1:
            return node.data + helper(node.left, level + 1) + helper(node.right, level + 1)
        else:
            return -node.data + helper(node.left, level + 1) + helper(node.right, level + 1)
  return helper(root, 1)
