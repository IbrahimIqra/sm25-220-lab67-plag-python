def swap_child(root, level, M):
  if root==None:
    return None
  if level<M:
    temp=root.left
    root.left=root.right
    root.right=temp
  swap_child(root.left, level+1, M)
  swap_child(root.right, level+1, M)
  return root

def level_sum(root):
  sum_dict={"even":0, "odd":0}
  #level=0
  def helper(node,level):
    if node==None:
      return 
    if level%2==0:
      sum_dict["even"]+=node.elem
    else:
      sum_dict["odd"]+=node.elem

    helper(node.left,level+1)
    helper(node.right,level+1)
  helper(root,0)
  return sum_dict["odd"]- sum_dict["even"]

def rangeSum(root, low, high):
  if root==None:
    return 0 

  if root.elem<low: 
    return rangeSum(root.right, low, high)

  if root.elem> high:
    return rangeSum(root.left, low, high)

  elif root.elem>=low and root.elem<=high: 
    return root.elem+ rangeSum(root.left, low,high)+ rangeSum(root.right,low,high)

def mirror_sum(root):


  if root==None: 
    return 0
  def helper(node1, node2):
    if node1==None or node2==None: 
      return 0 

    return (node1.elem+ node2.elem)+ helper(node1.left, node2.right)+ helper(node1.right, node2.left)

  return helper(root.left, root.right)

    node= node.right
  return node.elem

def helper_min(node):
   while node.left!=None:
    node= node.left
   return node.elem
  
def isBST(root):
    
  if root==None:
    return True
  if root.elem>= root.left.elem or root.elem<=root.right.elem: 
    return "False"

  if helper_max(root.right) <= root.elem  or helper_min(root.left)>= root.elem :
