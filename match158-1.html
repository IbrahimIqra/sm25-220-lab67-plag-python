    if(node==None):
      return(0,0)
    (leftOdd,leftEven)=(helper(node.left,level+1))
    (rightOdd,rightEven)=(helper(node.right,level+1))
    if(level%2==1):
      return(node.elem+leftOdd+rightOdd,leftEven+rightEven)

      return(leftOdd+rightOdd,node.elem+leftEven+rightEven)

    return True
