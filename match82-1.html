def swap_child(root, level, M):
    if root is None:
        return



    if level == M:
        root.left, root.right = root.right, root.left

    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)

    return root

def subtract_summation(root):


    def sum_subtree(node):
        if node is None:
            return 0
        return node.elem + sum_subtree(node.left) + sum_subtree(node.right)

    left_sum = sum_subtree(root.left) if root.left else 0
    right_sum = sum_subtree(root.right) if root.right else 0

    return left_sum - right_sum

def level_sum(root):
    if root is None:
        return 0

    queue = [None] * 1000
    head = 0
    tail = 0
    queue[tail] = root
    tail += 1

    level = 0
    diff = 0

    while head < tail:
        level_size = tail - head
        level_sum = 0

        for _ in range(level_size):
            node = queue[head]
            head += 1

            level_sum += node.elem

            if node.left is not None:
                queue[tail] = node.left
                tail += 1
            if node.right is not None:
                queue[tail] = node.right
                tail += 1

        if level % 2 == 0:
            diff -= level_sum
        else:
            diff += level_sum

        level += 1

    return diff

def rangeSum(root, low, high):
    if root is None:
        return 0

    if root.elem < low:
        return rangeSum(root.right, low, high)

    if root.elem > high:
        return rangeSum(root.left, low, high)

    return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum_helper(left, right):
    if left is None or right is None:
        return 0
    return left.elem + right.elem + mirror_sum_helper(left.left, right.right) + mirror_sum_helper(left.right, right.left)

def mirror_sum(root):
    if root is None:
        return 0
    return root.elem + mirror_sum_helper(root.left, root.right)

def isBST_helper(root, min_val, max_val):
    if root is None:
        return True
    if root.elem <= min_val or root.elem >= max_val:
        return False
    return isBST_helper(root.left, min_val, root.elem) and isBST_helper(root.right, root.elem, max_val)

def isBST(root):
    return isBST_helper(root, float('-inf'), float('inf'))
