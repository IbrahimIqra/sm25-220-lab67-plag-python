def swap_child(root, level, M):
    if root is None:
        return root
    if level >= M:
        root.left, root.right = root.right, root.left
    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)
    return root

def subtract_summation(root):
    def subtree_sum(node):
        if node is None:
            return 0
        return node.elem + subtree_sum(node.left) + subtree_sum(node.right)
    if root is None:
        return 0


    left_sum = subtree_sum(root.left)
    right_sum = subtree_sum(root.right)
    return left_sum - right_sum

def level_sum(root, level=1):
    if root is None:
        return 0
    current_val = root.elem if level % 2 == 1 else -root.elem
    return current_val + level_sum(root.left, level + 1) + level_sum(root.right, level + 1)

def rangeSum(root, low, high):
    if root is None:
        return 0
    if root.elem < low:
        return rangeSum(root.right, low, high)
    if root.elem > high:
        return rangeSum(root.left, low, high)
    return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
    def mirror_pair_sum(left, right):
        if left is None or right is None:
            return 0
        total = left.elem + right.elem
        total += mirror_pair_sum(left.left, right.right)
        total += mirror_pair_sum(left.right, right.left)
        return total
    if root is None:
        return 0
    return mirror_pair_sum(root.left, root.right)

def isBST(root):
    prev = [None]
    def inorder(node):
        if node is None:
            return True
        if not inorder(node.left):
            return False
        if prev[0] is not None and node.elem <= prev[0]:
            return False
        prev[0] = node.elem
        return inorder(node.right)
    return inorder(root)
