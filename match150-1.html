def level_sum(root):
    if not root:
        return 0

    def sum_helper(node, level, odd_sum, even_sum):
        if not node:
            return odd_sum, even_sum

        if level % 2 != 0:
            odd_sum += node.elem
        else:
            even_sum += node.elem

        left_odd, left_even = sum_helper(node.left, level + 1, odd_sum, even_sum)
        right_odd, right_even = sum_helper(node.right, level + 1, left_odd, left_even)

        return right_odd, right_even

    final_odd, final_even = sum_helper(root, 0, 0, 0)
    return final_odd - final_even

def rangeSum(root, low, high):
   if not root:
    return 0

   sum=0
   if low<=root.elem<=high:
    sum+=root.elem

   if root.elem<high:
    sum+=rangeSum(root.right,low,high)

   if root.elem>low:
    sum+=rangeSum(root.left,low,high)

   return sum

def mirror_sum(root):
   def checkMirror(left,right):
      if not left and not right:
        return 0
      if not left or not right:
        return 0

      current=left.elem+right.elem
      outer=checkMirror(left.left,right.right)
      inner=checkMirror(left.right,right.left)

      return current+outer+inner

   return checkMirror(root.left,root.right)
