def swap_child(root, level, M):


  if root == None:
        return None
  if level >= M:
        temp = root.left
        root.left = root.right
        root.right = temp
  swap_child(root.left, level+1, M)
  swap_child(root.right, level+1, M)

  return root

  if root == None:
        return 0

    if root == None:
        return 0

    right_sum = sum_subtree(root.right)

def level_sum(root):
  if root == None:
        return 0
  arr = [None] * 100
  front= 0
  prev = 0
  arr[prev] = (root, 0)
  sum = 0
  while front <= prev:
        node, level = arr[front]
        front += 1
        if level == 0:
            sum += node.elem
        else:
            sum += node.elem % level
        if node.left != None:
            prev += 1
            arr[prev] = (node.left, level + 1)
        if node.right != None:
            prev += 1
            arr[prev] = (node.right, level + 1)
  return sum

def rangeSum(root, low, high):


    if root == None:
        return 0
    total = 0
    if root.elem >= low and root.elem <= high:
        total = total + root.elem
    if root.elem > low:
        total = total + rangeSum(root.left, low, high)
    if root.elem < high:
        total = total + rangeSum(root.right, low, high)
    return total

def mirror_sum(root):
    if root == None:
        return 0
    def helper(left, right):
        if left == None or right == None:


            return 0
        total = left.elem + right.elem
        total = total + helper(left.left, right.right)
        total = total + helper(left.right, right.left)
        return total
    return helper(root.left, root.right)
