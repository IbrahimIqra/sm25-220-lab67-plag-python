def swap_child(root, level, M):
  if root == None:
    return None
  if level >= M:
    temp = root.left
    root.left = root.right
    root.right = temp
  else:


    swap_child(root.left, level+1, M)
    swap_child(root.right, level+1, M)
  return root

def subtract_summation(root):
    if root is None:
        return 0
    left_sum = subtree_sum(root.left)
    right_sum = subtree_sum(root.right)
    return left_sum - right_sum
def subtree_sum(node):
    if node is None:
        return 0
    return node.elem + subtree_sum(node.left) + subtree_sum(node.right)

def level_sum(root):
    even_sum = [0]
    odd_sum = [0]
    
    def dfs(node, level):
        if not node:
            return
        if level % 2 == 0:
            even_sum[0] += node.elem
        else:
            odd_sum[0] += node.elem
        dfs(node.left, level + 1)
        dfs(node.right, level + 1)
    
    dfs(root, 0)
    return odd_sum[0] - even_sum[0]

def rangeSum(root, low, high):
    if root is None:
        return 0
    total = 0
    if low <= root.elem <= high:
        total += root.elem
    if root.elem > low:
        total += rangeSum(root.left, low, high)
    if root.elem < high:
        total += rangeSum(root.right, low, high)
    return total

def mirror_sum(root):
    if root is None:
        return 0
    def sum_mirror_nodes(left, right):
        if left is None or right is None:
            return 0
        return left.elem + right.elem + sum_mirror_nodes(left.left, right.right) + sum_mirror_nodes(left.right, right.left)
    return sum_mirror_nodes(root.left, root.right)

def isBST(root):
    def is_bst_helper(node, min_val, max_val):
        if node is None:
            return True
        if min_val is not None and node.elem <= min_val:
            return False
        if max_val is not None and node.elem >= max_val:
            return False
        return (is_bst_helper(node.left, min_val, node.elem) and 
                is_bst_helper(node.right, node.elem, max_val))
    return is_bst_helper(root, None, None)
