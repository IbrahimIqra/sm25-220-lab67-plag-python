def level_sum(root):
    def solver(node, level):
        if node == None:
            return 0
        if level % 2 == 0:
            return -node.elem + solver(node.left, level+1) + solver(node.right, level+1)
        else:
            return node.elem + solver(node.left, level+1) + solver(node.right, level+1)
    return solver(root, 0)

def rangeSum(root, low, high):
    if root == None:


        return 0

    total_sum = 0

    if low <= root.elem <= high:
        total_sum += root.elem
    if root.elem > low:
        total_sum += rangeSum(root.left, low, high)
    if root.elem < high:
        total_sum += rangeSum(root.right, low, high)

    return total_sum

def mirror_sum(root):
    if root == None:
        return 0

    def get_mirror_sum_helper(left_node, right_node):
        if left_node == None and right_node == None:
            return 0
        if left_node == None or right_node == None:
            return 0
        current_sum = left_node.elem + right_node.elem
        left_mirror_sum = get_mirror_sum_helper(left_node.left, right_node.right)
        right_mirror_sum = get_mirror_sum_helper(left_node.right, right_node.left)

        return current_sum + left_mirror_sum + right_mirror_sum
    return get_mirror_sum_helper(root.left, root.right)

def isBST(root):
    def validate_bst(node, min_val, max_val):
        if node == None:
            return True


        if node.elem <= min_val or node.elem >= max_val:
            return False
        return (validate_bst(node.left, min_val, node.elem) and validate_bst(node.right, node.elem, max_val))
    return validate_bst(root, float('-inf'), float('inf'))
