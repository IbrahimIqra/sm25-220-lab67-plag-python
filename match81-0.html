def swap_child(root, level, M):
  if root is None:
    return None
  if level <M:
    temp=root.left
    root.left=root.right
    root.right=temp
 


  swap_child(root.left,level+1,M)
  swap_child(root.right,level+1,M)
  return root

def rangeSum(root, low, high):
  if root is None:
    return 0
  sum=0
  if root.elem>=low and root.elem<=high:
    sum=sum+ root.elem


  if root.elem>low:
    sum=sum+rangeSum(root.left,low,high)
  if root.elem<high:


    sum=sum+rangeSum(root.right,low,high)
  return sum

def isBST(root):
  return helper(root,min_val=None,max_val=None)


def helper(root,min_val,max_val):
  if root is None:
    return True
  if min_val is not None and root.elem<=min_val:
    return False
  if max_val is not None and root.elem>=max_val:
    return False
 
  left_bst=helper(root.left,min_val,root.elem)


  right_bst=helper(root.right,root.elem,max_val)


  return left_bst and right_bst
