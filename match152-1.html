def rangeSum(root, low, high):


    if root is None:
        return 0
    if root.val < low:
        return rangeSum(root.right, low, high)
    if root.val > high:
        return rangeSum(root.left, low, high)
    return root.val + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
    if root is None:
        return 0
    
    def helper(left_node, right_node):
        if left_node is None or right_node is None:
            return 0
        
        mirror_sum = left_node.elem + right_node.elem
        
        mirror_sum += helper(left_node.left, right_node.right)
        mirror_sum += helper(left_node.right, right_node.left)
        
        return mirror_sum
    
    return helper(root.left, root.right)

  def isBST(root):
    def helper(node, min, max):
        if node is None:
            return True
        if min is not None and node.elem <= min:
            return False
        if max is not None and node.elem >= max:
            return False
        
        return (helper(node.left, min, node.elem) and 
                helper(node.right, node.elem, max))
    
    return helper(root, None, None)
