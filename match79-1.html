def swap_child(root, level, M):
    if root is None:
        return None

    if level >= M:


        root.left, root.right = root.right, root.left
    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)
    return root

def subtract_summation(root):
    if root is None:
        return 0

    def subtree_sum(node):
        if node is None:
            return 0
        return node.elem + subtree_sum(node.left) + subtree_sum(node.right)

    left_sum = subtree_sum(root.left)
    right_sum = subtree_sum(root.right)

    return left_sum - right_sum

        else:

    return sum(root, 1)

def rangeSum(root, low, high):
  if root== None:
    return 0
  elif root.elem <low:
    return rangeSum(root.right, low, high)
  elif root.elem >high:
    return rangeSum(root.left, low, high)
  else:
    return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):


    if root== None:
        return 0

    def sum(left, right):
        if left== None or right== None:
            return 0
        return left.elem + right.elem + sum(left.left, right.right) + sum(left.right, right.left)

    return sum(root.left, root.right)

def isBST(root):
    def finder(node, min_val, max_val):
        if node==None:
            return True
        if not (min_val < node.elem < max_val):
            return False
        return finder(node.left, min_val, node.elem) and finder(node.right, node.elem, max_val)
    return finder(root, float('-inf'), float('inf'))
