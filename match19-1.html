    if root is None:
        return
    if level < M:
        root.left, root.right = root.right, root.left
    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)
    return root

def level_sum(root, level=0):


  if root is None:
    return 0


  if level%2==0:
    curr=-root.elem
  else:
    curr=root.elem


  l=level_sum(root.left,level+1)
  r=level_sum(root.right,level+1)


  return curr+l+r

def rangeSum(root, low, high,total=0):
  if root is None:
    return total


  if low<=root.elem<=high:
    total+=root.elem


  total=rangeSum(root.left,low,high,total)
  total=rangeSum(root.right,low,high,total)


  return total

def mirror_sum_helper(left, right):
    if left is None or right is None:
        return 0


    total = left.elem + right.elem


    total += mirror_sum_helper(left.left, right.right)
    total += mirror_sum_helper(left.right, right.left)
    return total


def mirror_sum(root):


    if root is None:
        return 0


    return mirror_sum_helper(root.left, root.right)

def isBST(root):
    def isBST_helper(node, min_val, max_val):
        if node is None:
            return True


        if min_val is not None and node.elem <= min_val:
            return False


        if max_val is not None and node.elem >= max_val:
            return False


        return (isBST_helper(node.left, min_val, node.elem) and
                isBST_helper(node.right, node.elem, max_val))


    return isBST_helper(root, None, None)
