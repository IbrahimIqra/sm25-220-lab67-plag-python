def swap_child(root, level, M):
  #To Do

    return None

  if 0<=level<M:



    nod.left=swap_child(root.right, level+1, M)
    nod.right=swap_child(root.left, level+1, M)
  else:
    nod.left=swap_child(root.left, level+1, M)
    nod.right=swap_child(root.right, level+1, M)
  return nod

    return sum

def rangeSum(root, low, high):
  #To Do
  if not root:
    return 0
  current_sum = 0

  if low <= root.elem <= high:


    current_sum += root.elem

  if root.elem > low:
    current_sum += rangeSum(root.left, low, high)

  if root.elem < high:
    current_sum += rangeSum(root.right, low, high)

  return current_sum



5.Mirror Sum of a BST::

def mirror_sum(root):
  #To Do
    if not root:
        return 0
    total = 0
    nodes = [root.left,root.right]
    while nodes:
        new_nodes =[]
        for i in range(0, len(nodes), 2):
            left=nodes[i]
            right=nodes[i+1]
            if left and right:


                total+=left.elem+right.elem
                new_nodes.append(left.left)
                new_nodes.append(right.right)
                new_nodes.append(left.right)
                new_nodes.append(right.left)
        nodes=new_nodes
    return total
