def swap_child(root, level, M):
    if root is None:
        return

    if level >= M:
        temp = root.left
        root.left = root.right


        root.right = temp

    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)

    return root

def level_sum(root):
    def helper(node, level):

        if node is None:
            return 0

        if level % 2 == 0:
            return -node.elem + helper(node.left, level + 1) + helper(node.right, level + 1)

        else:
            return node.elem + helper(node.left, level + 1) + helper(node.right, level + 1)
    
    return helper(root, 0)

def rangeSum(root, low, high):
    if root is None:
        return 0

    if root.elem < low:
        return rangeSum(root.right, low, high)

    if root.elem > high:
        return rangeSum(root.left, low, high)

    return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
    if root is None:
        return 0

    def mirrorHelper(left, right):


        if left is None or right is None:
            return 0

        return (left.elem + right.elem +
                mirrorHelper(left.left, right.right) +
                mirrorHelper(left.right, right.left))

    return mirrorHelper(root.left, root.right)

def isBST(root):
    def helper(node, low, high):
        if node is None:
            return True



        if not (low < node.elem < high):
            return False
            
        return helper(node.left, low, node.elem) and helper(node.right, node.elem, high)

    return helper(root, float('-inf'), float('inf'))
