def swap_child(root, level, M):
  if root is None:
    return root
  if level % M == 0:
    root.left, root.right = root.right, root.left
  swap_child(root.left, level+1, M)
  swap_child(root.right, level+1, M)
  return root

def level_sum(root):
  def current_lvl(node, level):
    if node is None:
      return 0
    if level % 2 == 0:
      return current_lvl(node.left, level+1) + current_lvl(node.right, level+1) + node.elem
    else:
      return current_lvl(node.left, level+1) + current_lvl(node.right, level+1) -node.elem
  return current_lvl(root, 1)

def rangeSum(root, low, high):
  if root is None:
    return 0
  if root.elem<low:
    return rangeSum(root.right, low, high)
  if root.elem>high:
    return rangeSum(root.left, low, high)
  return root.elem+ rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
  if root is None:
    return 0


  def helper(left, right):
    if left is None or right is None:
      return 0
       
    total = left.elem + right.elem
    total += helper(left.left, right.right)
    total += helper(left.right, right.left)
    return total


  return helper(root.left, root.right)

def isBST(root):
    prev = [None]
    def inorder_check(node):
        if node is None:
            return True
        if not inorder_check(node.left):
            return False
        if prev[0] is not None and node.elem <= prev[0]:
            return False
        prev[0] = node.elem
        return inorder_check(node.right)


    return inorder_check(root)
