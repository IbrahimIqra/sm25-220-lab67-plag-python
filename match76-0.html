  #To Do
  if root==None:
    return 0
  if root.left is None  and root.right is None:
    return root.elem
  else:
    if level<M:
      temp=root.left
      root.left=root.right
      root.right=temp
      swap_child(root.left,level+1,M)


      swap_child(root.right,level+1,M)
      return root

def rangeSum(root, low, high):
  #To Do
  if root == None:
    return 0
 
  curr=0
  if low<=root.elem<=high:
    curr=root.elem
  lsum=0
  if root.elem>low:
    lsum=rangeSum(root.left, low, high)
   
  rsum=0
  if root.elem<high:
    rsum=rangeSum(root.right, low, high)
 
  return curr+lsum+rsum

def mirror_sum(root):
  #To Do


  if root==None:
    return 0


  def helper(leftnode,rightnode):


    if leftnode==None or rightnode==None:
      return 0
   
    initialsum=leftnode.elem+rightnode.elem


    lsum= helper(leftnode.left,rightnode.right)


    rsum=helper(leftnode.right, rightnode.left)
   
    return initialsum + lsum + rsum
 
  return helper(root.left, root.right)
