def swap_child(root, level, M):
  
  if root is None or level> M:
    return root
  
  else:
    swap_child(root.left, level+1, M)
    swap_child(root.right, level+1, M)
  if level < M:
    root.left, root.right =root.right, root.left

  return root

    return 0

      return 0
    else:
      if level%2 != 0:

def rangeSum(root, low, high):

  if root == None:
    return 0
  
  if root.elem> high:
    return rangeSum(root.left, low, high)
  
  if low > root.elem:
    return rangeSum(root.right, low, high)



  return root.elem+rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):

  def helper(left, right):


    if left == None or right == None:
      return 0
    return left.elem + right.elem + helper(left.left, right.right) +helper(left.right, right.left)

  
  if root == None:
    return 0


  return helper(root.left, root.right)

def isBST(root):

    if root is None:
        return True

    def findMax(node):
      if node.right== None:
        return node.elem
      return findMax(node.right)
    
    def findMin(node):
      if node.left ==None:
          return node.elem
      return findMin(node.left)

    if root.left and findMax(root.left)>= root.elem:
        return False
    
    if root.right and findMin(root.right) <=root.elem:
        return False

    return True
