def swap_child(root, level, M):
  if root is None:
    return

  if level >= M:
    root.left, root.right = root.right, root.left

  swap_child(root.left, level + 1, M)
  swap_child(root.right, level + 1, M)
  return

def subtract_summation(root):

  def subtree_sum(node):

    if node is None:
      return 0

    return subtree_sum(node.left) + subtree_sum(node.right) + node.elem
    
  return subtree_sum(root.left) - subtree_sum(root.right)

def level_sum(root):
    if not root:
        return 0

    q = [(root, 1)]  
    f = 0
    o_sum = 0
    e_sum = 0

    while f < len(q):
        current, level = q[f]
        f += 1



        if level % 2 == 1:
            o_sum += current.elem
        else:
            e_sum += current.elem

        if current.left:
            q.append((current.left, level + 1))
            
        if current.right:
            q.append((current.right, level + 1))

    return e_sum - o_sum

def rangeSum(root, low, high):

  if root is None:
    return 0
  t = 0

  if root.elem >= low and root.elem <= high:
    t += root.elem

  if root.elem > low:
    t += rangeSum(root.left, low, high)

  if root.elem < high:
    t += rangeSum(root.right, low, high)

  return t

def mirror_sum(root):
  def helper(n1, n2):

    if n1 is None or n2 is None:
      return 0
    return n1.elem + n2.elem + helper(n1.left, n2.right) + helper(n1.right, n2.left)

  if root is None:
    return 0
  return helper(root.left, root.right)

def isBST(root):
  


  def helper(node, low, high):

    if node is None:
      return True

    if node.elem <= low or node.elem >= high:
      return False

    return helper(node.left, low, node.elem) and helper(node.right, node.elem, high)

  return helper(root, float('-inf'), float('inf'))
