def swap_child(root, level, M):
    if root is None:
        return root

    if level >= M:
        root.left, root.right = root.right, root.left

    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)

def subtract_summation(root):
    if root is None:
        return 0

    def sum(n):
        if n is None:
            return 0
        return n.elem+sum(n.left)+sum(n.right)

    return sum(root.left)-sum(root.right)

def level_sum(root, l=1):
    if root is None:
        return 0

    if l%2==1:
        return root.elem+level_sum(root.left,l+1)+level_sum(root.right, l + 1)
    else:
        return -root.elem+ level_sum(root.left,l+1)+level_sum(root.right, l + 1)

def rangeSum(root,low,high):
    if root is None:
        return 0


    if root.elem<low:
        return rangeSum(root.right,low,high)


    if root.elem>high:
        return rangeSum(root.left,low,high)


    return root.elem+rangeSum(root.left,low,high)+rangeSum(root.right,low,high)

def mirror_sum(root):
    if root is None:
        return 0

    def dummy(ln,rn):
        if ln is None or rn is None:
            return 0

        return (ln.elem+rn.val+dummy(ln.left, rn.right)+dummy(ln.right, rn.left))

    return dummy(root.left,root.right)

def isBST(root):
    a = []
    def in_traversal(node):
        if node:
            in_traversal(node.left)
            a.append(node.val)
            in_traversal(node.right)
    in_traversal(root)
    for j in range(1,len(a)):
        if a[j]<=a[j-1]:
            return False
    return True
