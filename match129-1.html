def subtree(root):
  if root == None:
    return 0
  return root.elem + subtree(root.left) + subtree(root.right)
def subtract_summation(root):
  if root == None:
    return 0
  left_tree = subtree(root.left)
  right_tree = subtree(root.right)
  return left_tree-right_tree

def level_sum(root, lvl = 1):
  if root == None:
    return 0


  if lvl % 2 == 1:
    return root.elem + level_sum(root.left,lvl+1) + level_sum(root.right,lvl+1)
  else:
    return -root.elem + level_sum(root.left,lvl+1) + level_sum(root.right,lvl+1)

def mirrorsum(left,right):


  if left == None or right == None:
    return 0
  total = left.elem + right.elem
  total+=mirrorsum(left.left,right.right)
  total+=mirrorsum(left.right,right.left)
  return total
