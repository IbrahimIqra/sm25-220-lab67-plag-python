class BTNode:
    def __init__(self, elem):
        self.elem = elem
        self.right = None
        self.left = None


def tree_construction(arr, i=1):
    if i >= len(arr) or arr[i] is None:
        return None
    p = BTNode(arr[i])
    p.left = tree_construction(arr, 2 * i)
    p.right = tree_construction(arr, 2 * i + 1)
    return p

def level_sum(root):
  if root == None:
    return 0

  queue = [None] * 100  # Fixed array for BFS
  start = 0
  end = 1
  queue[0] = root

  level = 1    odd = 0
  even = 0

  while start < end:
    count = end - start  # Number of nodes in this level
    temp_sum = 0

    for i in range(count):
      node = queue[start]
      start = start + 1
      temp_sum = temp_sum + node.elem

      if node.left != None:
        queue[end] = node.left
        end = end + 1
      if node.right != None:
        queue[end] = node.right
        end = end + 1

    if level % 2 == 1:
      odd = odd + temp_sum
    else:
      even = even + temp_sum

    level = level + 1



  return odd - even

def rangeSum(root, low, high):
  if root == None:   # if tree is empty
    return 0
  
  total = 0
  
 
  if root.elem >= low and root.elem <= high:
    total = total + root.elem
    if root.elem > low:
    total = total + rangeSum(root.left, low, high)
  

  if root.elem < high:
    total = total + rangeSum(root.right, low, high)

  return total

def mirror_sum(root):
  if root == None:
    return 0
  def helper(left, right):
    if left == None or right == None:
      return 0

       total = left.elem + right.elem

    
    total = total + helper(left.left, right.right)
    total = total + helper(left.right, right.left)

    return total

  return helper(root.left, root.right)

def checkBST(node, mini, maxi):


    if node == None:
        return True
    if node.elem <= mini or node.elem >= maxi:
        return False
    return checkBST(node.left, mini, node.elem) and checkBST(node.right, node.elem, maxi)

def isBST(root):
    return checkBST(root, float("-inf"), float("inf"))
