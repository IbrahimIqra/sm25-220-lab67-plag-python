def swap_child(root, level, M):
    if root is None:
        return None


   
    if level >= M:
        root.left, root.right = root.right, root.left


   
    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)


    return root

def subtract_summation(root):
    def subtree_sum(node):
        if node is None:
            return 0
        return node.elem + subtree_sum(node.left) + subtree_sum(node.right)


    if root is None:
        return 0


    return subtree_sum(root.left) - subtree_sum(root.right)

def level_sum(root):
    def dfs(node, level):
        if node is None:
            return 0
       
        if level % 2 == 1:
            return node.elem + dfs(node.left, level + 1) + dfs(node.right, level + 1)
        else:
            return -node.elem + dfs(node.left, level + 1) + dfs(node.right, level + 1)
   
    return dfs(root, 1)

def rangeSum(root, low, high):
    if not root:
        return 0


    if root.elem < low:
        return rangeSum(root.right, low, high)
    elif root.elem > high:
       
        return rangeSum(root.left, low, high)
    else:
        
        return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
    def sum_mirror_nodes(node1, node2):


        if not node1 or not node2:
            return 0
        return (node1.elem + node2.elem +
                sum_mirror_nodes(node1.left, node2.right) +
                sum_mirror_nodes(node1.right, node2.left))


    if not root:
        return 0
    return sum_mirror_nodes(root.left, root.right)

def isBST(root):
    def is_valid(node, min_val, max_val):
        if not node:
            return True


        if not (min_val < node.elem < max_val):
            return False
        return (is_valid(node.left, min_val, node.elem) and
                is_valid(node.right, node.elem, max_val))


    return is_valid(root, float('-inf'), float('inf'))
