def swap_child(root, level, M):
  #To Do
  if root==None:
    return


  if level < M:
    sum= root.left
    root.left= root.right
    root.right= sum

  swap_child(root.left, level + 1, M)
  swap_child(root.right, level + 1, M)

  return root

def level_sum(root):
    if root == None:
        return 0
    
    def solve(node, level):
        if node == None:
            return 0
        temp1= solve(node.left, -level)
        temp2= solve(node.right, -level)
        return level * node.elem + temp1 + temp2
    return solve(root, -1)

def rangeSum(root, low, high):
    total = 0    
    if root == None:
        return 0
    if low <= root.elem <= high:
        total += root.elem
    total += rangeSum(root.left, low, high)
    total += rangeSum(root.right, low, high)

    return total

def mirror_sum(root):


    if root == None:
        return 0

    def helper(left, right):
        if left == None or right == None:
            return 0
        total = left.elem + right.elem
        total += helper(left.left, right.right)
        total += helper(left.right, right.left)
        return total
    return helper(root.left, root.right)

def isBST(root):
    prev = [None]
    def inorder(node):
        if node == None:
            return True
        if not inorder(node.left):
            return False
        if prev[0] != None and node.elem <= prev[0]:
            return False
        prev[0] = node.elem
        return inorder(node.right)

    return inorder(root)
