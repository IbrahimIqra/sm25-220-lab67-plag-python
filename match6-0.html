def swap_child(root, level, M):
    if root is None:
        return


    # Recurse into subtrees to reach the level M
    if level >= M:
        # Swap children nodes


        root.left, root.right = root.right, root.left


    # Recursively call for left and right subtrees, incrementing level
    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)


    return root

def subtract_summation(root):
    if not root:
        return 0
    # Compute sums of the left and right subtrees
    left_sum = subtree_sum(root.left)
    right_sum = subtree_sum(root.right)
    # Return the difference of the left and right subtree sums
    return left_sum - right_sum


def subtree_sum(node):
    if not node:
        return 0
    # Recursively sum all values in the subtree


    return node.elem + subtree_sum(node.left) + subtree_sum(node.right)

def rangeSum(root, low, high):
    if root == None:
        return 0


    if root.elem < low:
        return rangeSum(root.right, low, high)


    if root.elem > high:
        return rangeSum(root.left, low, high)


    return (root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high))

def mirror_sum(root):
    def helper(l, r):
        if l== None or r== None:
            return 0


        sum = l.elem + r.elem


        sum += helper(l.left, r.right)
        sum += helper(l.right, r.left)


        return sum


    if root == None:
        return 0


    return helper(root.left, root.right)

def isBST(root):
    def helper(node, min_v, max_v):
        if node == None:
            return True
        if min_v != None and node.elem <= min_v:
            return False
        if max_v != None and node.elem >= max_v:
            return False
        return (helper(node.left, min_v, node.elem) and
                helper(node.right, node.elem, max_v))


    return helper(root, None, None)
