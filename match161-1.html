def swap_child(root, level, M):


    if root is None:
        return None
    if level >= M:
         x=root.left
         root.left=root.right
         root.right=x
    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)
    return root

def subtract_summation(root):
    if root is None:
        return 0
    left_sum = subtree_sum(root.left)
    right_sum = subtree_sum(root.right)
    return left_sum - right_sum


def subtree_sum(node):
    if node is None:
        return 0
    return node.elem + subtree_sum(node.left) + subtree_sum(node.right)

def rangeSum(root, low, high):
    if root is None:
        return 0




    if root.elem < low:
        return rangeSum(root.right, low, high)




    elif root.elem > high:
        return rangeSum(root.left, low, high)




    else:
        return (root.elem +
                rangeSum(root.left, low, high) +
                rangeSum(root.right, low, high))

def mirror_sum(root):
    if not root:
        return 0


    def helper(left, right):
        if not left or not right:
            return 0


        s = left.elem + right.elem
        s += helper(left.left, right.right)
        s += helper(left.right, right.left)
        return s


    return helper(root.left, root.right)

def isBST(root):
    def helper(node, min_val, max_val):
        if not node:
            return True


        if not (min_val < node.elem < max_val):
            return False
        return helper(node.left, min_val, node.elem) and helper(node.right, node.elem, max_val)


    return helper(root, float('-inf'), float('inf'))
