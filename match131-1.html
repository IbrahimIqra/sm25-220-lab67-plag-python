def swap_child(root, level, M):
    if root is None: return root
    if level >= M:
        root.left, root.right = root.right, root.left
    swap_child(root.left, level+1, M)
    swap_child(root.right, level+1, M)
    return root

def subtract_summation(root):
    if root is None:
        return 0
 
    def subtree_sum(node):
        if node is None:
            return 0
        return node.elem + subtree_sum(node.left) + subtree_sum(node.right)


    left_sum = subtree_sum(root.left)
    right_sum = subtree_sum(root.right)
    return left_sum - right_sum

def difference_of_level_sum(root, level=1):
    if root is None:
        return 0
    if level % 2 == 1:  
        return root.elem + difference_of_level_sum(root.left, level+1) + difference_of_level_sum(root.right, level+1)
    else:                
        return -root.elem + difference_of_level_sum(root.left, level+1) + difference_of_level_sum(root.right, level+1)

def rangeSum(root, low, high):
    if root is None:
        return 0
    if root.elem < low:
        return rangeSum(root.right, low, high)
    if root.elem > high:
        return rangeSum(root.left, low, high)
    return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
    def pair_sum(a, b):
        if a is None or b is None:
            return 0
        s = a.elem + b.elem
        s += pair_sum(a.left, b.right)
        s += pair_sum(a.right, b.left)
        return s
    if root is None:
        return 0
    return pair_sum(root.left, root.right)

def isBST(root):
    def ok(node, lo, hi):
        if node is None:
            return True
        if not (lo < node.elem < hi):
            return False
        return ok(node.left, lo, node.elem) and ok(node.right, node.elem, hi)
 
    return ok(root, float("-inf"), float("inf"))
