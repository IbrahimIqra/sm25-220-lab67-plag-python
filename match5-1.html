def level_sum(root):
  def sum(n,lvl):
    if n is None :
      return 0
    if lvl%2==0:
      val=-n.elem
    else:
      val=n.elem    
    left=sum(n.left,lvl+1)
    right=sum(n.right,lvl+1)        
    return val+left+right    
  return sum(root,0)

#task 4
def rangeSum(root, low, high):
    if root is None:
        return 0

    sum_val= 0           #eta return hobe prottekbar recursion hole
    if low  <= root.elem  <=high:                  #range er moddhe thakvbe
        sum_val+=  root.elem        
    if root.elem >  low:
        sum_val  += rangeSum( root.left,low, high )      # boro hole left pass 
    if  root.elem<  high:           #choto hole right recursion
        sum_val += rangeSum(root.right, low, high)
    return  sum_val
