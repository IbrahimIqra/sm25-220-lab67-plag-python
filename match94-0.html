def swap_child(root, level, M):
  if root!=None and  level< M:
    left= root.right
    right=root.left
    root.left =left
    root.right=right
  else:


    return root
  swap_child(root.left,level+1 ,M)
  swap_child(root.right,level +1,M)
  return root

def subtract_summation(root):
  if root==None:
    return 0
  def tree(node):
    if node is not None:
      sum= node.elem+ tree(node.left)+tree(node.right)
    else:
      return 0
    return sum
  sum1=tree(root.left)
  sum2=tree(root.right)
  subtraction=sum1-sum2


  return subtraction

def level_sum(root):
  def helper(node,level):
    if node ==None:
      return 0
    elif level%2==0:
      diff=node.elem+ helper(node.left,level+1)+ helper(node.right,level+1)
      return diff
    else:
      diff=helper(node.right,level+1)+helper(node.left,level+1)-node.elem
      return diff
  return helper(root,1)

    output=0

def mirror_sum(root):
  if root==None:
    return 0
  def helper(L,R):
    if L==None or R==None:
      return 0
    sum=L.elem+R.elem+ helper(L.left,R.right)+helper(R.left,L.right)
    return sum
  return helper(root.left,root.right)

def isBST(root):
  def bst(node,min,max):
    if node==None:
      return True
    if max!= None and node.elem>max:
      return False


    if min!= None and node.elem<min:
      return False
    return bst(node.left,min,node.elem) and bst(node.right, node.elem,max)
  return bst(root,None,None)
