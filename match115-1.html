def rangeSum(root, low, high):
  if root ==None:
    return 0
  sum=0
  if low <= root.elem <=high:
    sum+=root.elem
  if root.elem >=low:
    sum+=rangeSum(root.left, low,high)
  if root.elem <=high:
    sum+=rangeSum(root.right,low,high)


  return sum

def isBST(root):
  def assist(root, min_val, max_val):
    if root is None:
      return True
    if min_val is not None and root.elem <=min_val:
      return False


    if max_val is not None and root.elem >=max_val:
      return False
    return assist(root.left, min_val, root.elem) and assist(root.right, root.elem, max_val)
  return assist(root, None, None)
