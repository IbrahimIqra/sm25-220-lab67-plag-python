def swap_child(root, level, M):
    if root is None:
        return None
    if level == M:


        root.left, root.right = root.right, root.left

    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)
    return root

def subtract_summation(root):
    if root is None:
        return 0

    if root.left is None:
        right_sum = subtract_summation(root.right)
        root.elem = root.elem - right_sum
        return root.elem

    if root.right is None:
        left_sum = subtract_summation(root.left)
        root.elem = root.elem - left_sum
        return root.elem

    left_sum = subtract_summation(root.left)
    right_sum = subtract_summation(root.right)
    root.elem = root.elem - left_sum - right_sum


    return root.elem

def level_sum(root):
    if root is None:
        return 0
    line = [None] * 100
    sam = 0
    age = 1
    line[0] = (root, 0)  
    even=0
    odd=0

    while sam < age:
        node, level = line[sam]
        sam += 1
        if level % 2 == 0:
            even+= node.elem
        else:
            odd+= node.elem

        if node.left is not None:
            line[age] = (node.left, level + 1)
            age += 1
        if node.right is not None:
            line[age] = (node.right, level + 1)
            age += 1

    return odd-even

def rangeSum(root, low, high):
    if root is None:
        return 0
    
    if low <= root.elem <= high:
        return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)
    
    elif root.elem < low:
        return rangeSum(root.right, low, high)
    else:
        return rangeSum(root.left, low, high)

def mirror_sum(root):
  if root == None:
    return 0

  arr = [None]*100
  alv = 0
  sum = 0

  arr[alv] = (root.left, root.right)

  while alv >= 0:
    left, right = arr[alv]
    alv -= 1

    if left != None and right != None:
      sum += left.elem + right.elem

      alv += 1
      arr[alv] = (left.left, right.right)
      alv += 1
      arr[alv] = (left.right, right.left)

  return sum

def isBST(root):
    prev = [None]
    
    def inorder_check(root, prev):
        if root is None:
            return True
        
        if not inorder_check(root.left, prev):
            return False
        
        if prev[0] is not None and prev[0].elem >= root.elem:
            return False
        
        prev[0] = root
        
        return inorder_check(root.right, prev)
    
    return inorder_check(root, prev)
