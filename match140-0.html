def swap_child(root, level, M):
  if root is None:
    return None
  if root.left is None and root.right is None:
    return root


  if level >= M:
    root.left, root.right = root.right, root.left
  swap_child(root.left, level + 1, M)
  swap_child(root.right, level + 1, M)
  return root

def subtract_summation(root):
  def sumhelper(root):
    if root is None:
      return 0
    if root.left is None and root.right is None:
      return root.elem
    return root.elem + sumhelper(root.left) + sumhelper(root.right)
  return sumhelper(root.left) - sumhelper(root.right)

def level_sum(root):
  def level_sum_helper(root, level):
    if root is None:
      return 0
    curr = 0 
    if level % 2 == 1: 
      curr += root.elem
    else:  
      curr -= root.elem
    left_sum = level_sum_helper(root.left, level + 1)
    right_sum = level_sum_helper(root.right, level + 1)
        
    return cur + left_sum + right_sum
    
  return level_sum_helper(root, 1)

def rangeSum(root, low, high):
    if root is None:
        return 0
    if root.elem < low:
        return rangeSum(root.right, low, high)
    if root.elem > high:
        return rangeSum(root.left, low, high)
    result = root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)
    return result

def mirror_sum(root):
  if root is None:
    return 0
  def pair_sum(left_child, right_child):


    if left_child is None or right_child is None:
      return 0
    sum = left_child.elem + right_child.elem
    left_pair = pair_sum(left_child.left, right_child.right)
    right_pair = pair_sum(left_child.right, right_child.left)
    result = sum + left_pair + right_pair
    return result
  return pair_sum(root.left, root.right)
