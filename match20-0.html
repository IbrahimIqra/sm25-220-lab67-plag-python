def swap_child(root, level, M):


    if root is None:
        return None
   
    if level <= M:
        root.left, root.right = root.right, root.left
   
    if root.left:
        swap_child(root.left, level + 1, M)
    if root.right:
        swap_child(root.right, level + 1, M)
   
    return root

def level_sum(root):


    if root is None:
        return 0
   
    lvl_sums = {}
   
    def calculate_level_sums(node, level):
        if node is None:
            return
       
       
        if level in lvl_sums:
            lvl_sums[level] += node.elem
        else:
            lvl_sums[level] = node.elem
       
     
        calculate_level_sums(node.left, level + 1)
        calculate_level_sums(node.right, level + 1)


    calculate_level_sums(root, 0)


    odd_s = 0
    even_s = 0
   
    for level, sum in lvl_sums.items():
        if level % 2 == 0:  
            even_s += sum
        else:
            odd_s += sum
   
    return odd_s - even_s

def rangeSum(root, low, high):


    if root is None:
        return 0
   
    total = 0
   
    if low <= root.elem <= high:


        total+= root.elem


    if root.elem > low:
        total += rangeSum(root.left, low, high)
   
    if root.elem < high:
        total += rangeSum(root.right, low, high)
   
    return total

def isBST(root):


  def validate_bst(node, min, max):


      if node is None:
          return True
     


      if node.elem <= min or node.elem >= max:
          return False
     


      return (validate_bst(node.left, min, node.elem) and validate_bst(node.right, node.elem, max))
 
  return validate_bst(root, float('-inf'), float('inf'))
