def level_sum(root):
 def helper(val,level):
   if val == None:
     return (0,0)


   leftodd, lefteven = helper(val.left,level+1)
   rightodd, righteven = helper(val.right,level+1)


   if level% 2 == 1:
     return (val.elem + leftodd + rightodd, lefteven + righteven)
   else:
     return (leftodd + rightodd, val.elem + lefteven + righteven)

def rangeSum(root, low, high):
  if root == None:
       return 0


  if root.elem < low:
       return rangeSum(root.right, low, high)
  elif root.elem > high:
      
       return rangeSum(root.left, low, high)
  else:
      
       return (root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high))

def isBST(root):
 def helper(node, low, high):
       if node is None:
           return True
       # node must be strictly between low and high
       if not (low < node.elem < high):
           return False
       # check left subtree with new upper bound
       # check right subtree with new lower bound
       return helper(node.left, low, node.elem) and helper(node.right, node.elem, high)


       return helper(root, float('-inf'), float('inf'))
