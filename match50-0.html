def swap_child(root, level, M):
  if level == M:
    return root
  if root == None:
    return None

  left = swap_child(root.left,level+1,M)
  right = swap_child(root.right,level+1,M)

  root.right = left
  root.left = right

  return root

class BTNode:
    def __init__(self, val = 0):
        self.val = val
        self.left = None
        self.right = None



def rangeSum(root, low, high):

    if root == None:
        return 0
    
    if root.val < low:

        return rangeSum(root.right, low, high)

    if root.val > high:

        return rangeSum(root.left, low, high)
    
    return root.val + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
    """
    Calculates the sum of values of nodes that are mirrors of each other in a binary tree.
    """
    if not root or (not root.left and not root.right):
        return 0
    
    return helper(root.left, root.right)


def helper(node1, node2):

    if not node1 or not node2:
        return 0
    
    # If both nodes exist, they are mirror pairs
    current_sum = node1.val + node2.val
    
    # Recursive calls for inner and outer mirror pairs
    outer_sum = helper(node1.left, node2.right)
    inner_sum = helper(node1.right, node2.left)
    
    return current_sum + outer_sum + inner_sum
