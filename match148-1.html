def swap_child(root, level, M):
  if root == None:
    return None
  else:
    if level < M:
      templeft = swap_child(root.left,level+1,M)
      tempright = swap_child(root.right,level+1,M)
      root.left = tempright
      root.right = templeft
      return root
    else:
      return root

    sum = 0
    if  root == None:
      return 0
    else:
      top_value = root.elem
      sumleft = total_sum(root.left)
      sumright = total_sum(root.right)
      sum = top_value + sumleft + sumright
      return sum

  summation_left = total_sum(root.left)
  summation_right = total_sum(root.right)
  subtraction = summation_left - summation_right


  return subtraction

def level_sum(root):
  def sum_helper(root,level=0):
    curr_sum = 0
    if root == None:
      return 0 
    else:
      if level % 2 == 0:
        curr_sum += (-1)*root.elem
      elif level % 2 != 0:
        curr_sum += root.elem

    sumleft = sum_helper(root.left,level+1)
    sumright = sum_helper(root.right,level+1)
    curr_sum += sumleft + sumright
    return curr_sum
    
  levelsum = sum_helper(root)
  return levelsum

def rangeSum(root, low, high):
  sum = 0
  if root == None:
    return 0
  else:


    if root.elem < low:
      return rangeSum(root.right,low,high)
    if root.elem > high:
      return rangeSum(root.left,low,high)
    else:
      left = rangeSum(root.left,low,high)
      right = rangeSum(root.right,low,high)
      sum += root.elem + left + right
      return sum

def mirror_sum(root):
  def sum_helper(left,right):
    sum = 0
    if left == None or right == None:
      return 0
    else:
      sum1 = left.elem + right.elem
      sum2 = sum_helper(left.left,right.right)
      sum3 = sum_helper(left.right,right.left)
      sum = sum1 + sum2 + sum3
      return sum
  
  if root == None:
    return 0
  else:
    return sum_helper(root.left,root.right)

def isBST(root):
  if root == None:
    return None
  else:
    if root.left.elem < root.right.elem:
      return True
    else:
      return False
  
  isBST(root.left)
