def swap_child(root, level, M):
    if root is None:
        return 0

    if level <= M:
        root.left, root.right = root.right, root.left
    root.left = swap_child(root.left, level + 1, M)
    root.right = swap_child(root.right, level + 1, M)

    return root

def level_sum(root,x=1,y=0):
  if root == None:
    return y
  y = level_sum(root.left,x+1,y)
  if x%2 == 0:
    y+= root.elem

    y-= root.elem

def rangeSum(root, low, high):
    if root== None:


        return 0
    sum = 0    
    if low <= root.elem <= high:
        sum += root.elem
    
    if root.elem > low:
        sum += rangeSum(root.left, low, high)

    if root.elem < high:
        sum += rangeSum(root.right, low, high)
    
    return sum

def mirror_sum(root):
   if root is None:
      return 0
   return helper(root.left, root.right)
def helper(left,right):
  if left is None or right is None:
    return 0
  sum = left.elem + right.elem

  sum+= helper(left.left , right.right)
  sum+= helper(left.right , right.left)
  
  return sum

def isBST(root):
  def checker(node, min, max):
        if node is None:
            return True


        if node.elem <= min or node.elem >= max:
            return False
        return checker(node.left, min, node.elem) and checker(node.right, node.elem, max)
  return checker(root, -999, 999)
