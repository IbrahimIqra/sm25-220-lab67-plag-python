def swap_child(root, level, M):
  if root is None:
    return None
  if (level+1) % M == 0:
    root.left, root.right = root.right, root.left
  root.left = swap_child(root.left, level+1, M)


  root.right = swap_child(root.right, level+1, M)
  return root

def subtract_summation(root):
  def summation(node):
    if node is None:
      return 0
    return node.elem + summation(node.left) + summation(node.right)
  
  if root is None:
    return 0
  
  left_sum = summation(root.left)
  right_sum = summation(root.right)
  return left_sum - right_sum

def level_sum(root):
  result_sum = 0


  if root is None:
    return 0


  queue = [(root, 1)]


  while queue:
    current_node, current_level = queue.pop(0) 


    if current_level % 2 != 0:
      result_sum -= current_node.elem
    else: 
      result_sum += current_node.elem


    if current_node.left:
      queue.append((current_node.left, current_level + 1))
    if current_node.right:
      queue.append((current_node.right, current_level + 1))


  return result_sum

def rangeSum(root, low, high):
  if root == None:
    return 0


  total = 0


  if root.elem >= low and root.elem <= high:
    total += root.elem


  total += rangeSum(root.left, low, high)
  total += rangeSum(root.right, low, high)


  return total

def mirror_sum(root):


  if root == None:
    return 0


  def helper(left, right):
    if left == None or right == None:
      return 0


    total = left.elem + right.elem


    total += helper(left.left, right.right)
    total += helper(left.right, right.left)


    return total


  return helper(root.left, root.right)

def isBST(root):
  arr = []


  def inorder(node):
    if node == None:
      return
    inorder(node.left)
    arr.append(node.elem)
    inorder(node.right)


  inorder(root)


  for i in range(1, len(arr)):
    if arr[i] <= arr[i-1]:
      return False
  return True
