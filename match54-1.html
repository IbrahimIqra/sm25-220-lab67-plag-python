def swap_child(root, level, M):
    if root is None:
        return None


    NN = BTNode(root.elem)


    if level >= M:
        NN.left = swap_child(root.right, level + 1, M)
        NN.right = swap_child(root.left, level + 1, M)
    else:


        NN.left = swap_child(root.left, level + 1, M)
        NN.right = swap_child(root.right, level + 1, M)


    return NN

def level_sum(root):
    if root is None:
        return 0
   
    def lvldiff(node, lvl):
        if node is None:
            return 0        
       
        curr = node.elem
        if lvl % 2 == 1:
            curr = -curr
       
        lftsm = lvldiff(node.left, lvl + 1)
        rhtsm = lvldiff(node.right, lvl + 1)
        return curr + lftsm + rhtsm
   
    return lvldiff(root, 1)

def rangeSum(root, low, high):
    if root is None:
        return 0
   
    var = 0
    if low <= root.elem <= high:


        var = root.elem
    if root.elem > low:
        var += rangeSum(root.left, low, high)
   
    if root.elem < high:
        var += rangeSum(root.right, low, high)
   
    return var

def mirror_sum(root):
    if root is None:
        return 0
   
    def mirrorsumhlpr(lft, rht):
        if lft is None or rht is None:
            return 0


        current_sum = lft.elem + rht.elem
        lftsum = mirrorsumhlpr(lft.left, rht.right)
        rhtsum = mirrorsumhlpr(lft.right, rht.left)
       
        return current_sum + lftsum + rhtsum
   
    return mirrorsumhlpr(root.left, root.right)

def isBST(root):
    def isbsthlpr(node, varmin, varmax):
        if node is None:
            return True
       
        if varmin is not None and node.elem <= varmin:
            return False
        if varmax is not None and node.elem >= varmax:
            return False
       
        lft = isbsthlpr(node.left, varmin, node.elem)
        rht = isbsthlpr(node.right, node.elem, varmax)
       
        return lft and rht
   
    return isbsthlpr(root, None, None)
