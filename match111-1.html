def swap_child(root, level, M):














    if root is None:














        return None










    if level >= M:










        root.left, root.right = root.right, root.left










    swap_child(root.left, level + 1, M)








    swap_child(root.right, level + 1, M)
















    return root

def subtract_summation(root):


    if root is None:




        return 0










    def subtree_sum(node):






        if node is None:








            return 0






        return node.elem + subtree_sum(node.left) + subtree_sum(node.right)








    left_sum = subtree_sum(root.left)






    right_sum = subtree_sum(root.right)






    return left_sum - right_sum

def level_sum(root):








    def helper(node, level):










        if node is None:








            return 0






         
        if level % 2 == 1:








            return node.elem + helper(node.left, level + 1) + helper(node.right, level + 1)








        else:








            return -node.elem + helper(node.left, level + 1) + helper(node.right, level + 1)








    return helper(root, 1)

def rangeSum(root, low, high):






     
    if root is None:






        return 0








    v = root.elem   








    if v < low:






         
        return rangeSum(root.right, low, high)








    if v > high:






         
        return rangeSum(root.left, low, high)








     
    return v + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):








    if root is None:








        return 0








    def helper(left, right):








        if left is None or right is None:








            return 0








         
        return (left.elem + right.elem + helper(left.left, right.right) + helper(left.right, right.left))








    return helper(root.left, root.right)

def isBST(root):








    def helper(node, low, high):








        if node is None:








            return True








        if not (low < node.elem < high):








            return False








        return (helper(node.left, low, node.elem) and








                helper(node.right, node.elem, high))








    return helper(root, float("inf"), float("inf"))
