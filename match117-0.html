def rangeSum(root, low, high):
    if root==None:
      return 0
    result=0
       
    if high>root.elem:
        result += rangeSum(root.right, low, high)
   
    if low<root.elem:
        result += rangeSum(root.left, low, high)


    if high>= root.elem>=low:
        result += root.elem
   
    return result

def mirror_sum(root):
    if root==None:
      return 0
    def sum(left_nod, right_nod):
        if left_nod== None:
            return 0


        if right_nod== None:
          return 0
        total= left_nod.elem + right_nod.elem
        m= sum(left_nod.left, right_nod.right)
        n = sum(left_nod.right, right_nod.left)


        return total+m+n


    return sum(root.left, root.right)

def isBST(root):


    def check(nod, min, max):
        if nod == None:
            return True
   
        if max!=None and max.elem<=nod.elem:
            return False


        elif min!=None and min.elem>=nod.elem:
            return False


 
        check_left = check(nod.left, min, nod)
        check_right = check(nod.right, nod, max)
       
        return check_right and check_left
   
    return check(root, None, None)
