def level_sum(root, level=1):
    if root is None:
        return 0

    if level % 2 == 1:
        return -root.elem + level_sum(root.left, level+1) + level_sum(root.right, level+1)
    else:
        return root.elem + level_sum(root.left, level+1) + level_sum(root.right, level+1)

def rangeSum(root, low, high):


    if not root:
        return 0

    if root.elem < low:
        return rangeSum(root.right, low, high)

    if root.elem > high:
        return rangeSum(root.left, low, high)

    return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
    if not root:
        return 0

    def helper(left_node, right_node):
        if not left_node or not right_node:
            return 0



        current_sum = left_node.elem + right_node.elem

        current_sum += helper(left_node.left, right_node.right)
        current_sum += helper(left_node.right, right_node.left)

        return current_sum

    return helper(root.left, root.right)

def isBST(root):
    def helper(node, low, high):
        if not node:
            return True
        


        if not (low < node.elem < high):
            return False
        
        return helper(node.left, low, node.elem) and helper(node.right, node.elem, high)

    return helper(root, float('-inf'), float('inf'))
