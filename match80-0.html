def swap_child(root, level, M):
  if level == M:
    return root
  if root == None:
    return None
  left = swap_child(root.left,level+1,M)
  right = swap_child(root.right,level+1,M)
  root.right = left
  root.left = right
  return root

def level_sum(root):
  def helper(node, level):
    if not node:
      return 0
    l=helper(node.left, level + 1)
    r=helper(node.right, level + 1)
    if level % 2 != 0:
      return node.elem +l + r
    else:
      return -node.elem +l+ r
  return helper(root, 0)

def rangeSum(root, low, high):
  if root == None:
    return 0
  if root.elem < low:
    return rangeSum(root.right,low,high)
  if root.elem > high:
    return rangeSum(root.left,low,high)
  return root.elem + rangeSum(root.left,low,high) + rangeSum(root.right,low,high)

def mirror_sum(root):
  if root==None:
    return 0
  def helper(left, right):
    if left==None or right==None:
      return 0
    return left.elem + right.elem + helper(left.left, right.right) + helper(left.right, right.left)
  return helper(root.left, root.right)

def isBST(root):
    def helper(node, min_val, max_val):
        if not node:
            return True
        if (min_val!=None and node.elem<=min_val) or (max_val!=None and node.elem>=max_val):
            return False
        return helper(node.left,min_val,node.elem) and helper(node.right,node.elem,max_val)
    return helper(root, None, None)
