def rangeSum(root, low, high):
  sum=0
  if root is None:
    return 0



  if root.elem>low:
    sum+=rangeSum(root.left, low, high)

  if root.elem<high:
    sum+=rangeSum(root.right, low, high)

  if low<=root.elem<=high:
    sum+=root.elem

  return sum

def helper(left, right):
  if left is None or right is None:
    return 0

  C1=right.elem+left.elem
  C2=helper(left.right, right.left)
  C3=helper(left.left, right.right)

  return C1 + C2 + C3

def isBST(root):
    def helper(node, min, max):
        if node is None:
            return True
        if node.elem>=max or node.elem<=min:
            return False
        return helper(node.left, min, node.elem) and helper(node.right, node.elem, max)

    return helper(root, float('-inf'), float('inf'))
