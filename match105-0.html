def level_sum(root):
  if root==None:
    return 0
  def sum(root,l=0,sumd={}):


    if root==None:
      return
    if l%2==0:
      sumd["even"]+=root.elem
    else:
      sumd["odd"]+=root.elem
    sum(root.left,l+1,sumd)
    sum(root.right,l+1,sumd)
  sumd={"even":0,"odd":0}
  sum(root,0,sumd)
  return sumd["even"]-sumd["odd"]

def rangeSum(root, low, high):
  if root==None:
    return 0
  s=0
  if low<=root.elem<=high:
    s+=root.elem
  s+= rangeSum(root.left, low, high)
  s+= rangeSum(root.right, low, high)
  return s

def isBST(root):
  if root==None:
    return True
  if root.left!=None:
    left=findmax(root.left)
    if left>=root.elem:
      return False
  if root.right!=None:
    right=findmin(root.right)
    if right<=root.elem:
      return False
  return isBST(root.left) and isBST(root.right)
def findmax(root):
  if root.right==None:
    return root.elem
  else:
    return findmax(root.right)
def findmin(root):
  if root.left==None:
    return root.elem
  else:
    return findmin(root.left)
