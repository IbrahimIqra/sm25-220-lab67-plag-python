def swap_child(root, level, M):
    if root == None:
        return None


    if level < M:
        save = root.left      
        root.left = root.right
        root.right = save
     
    swap_child(root.left, level + 1, M)
    swap_child(root.right, level + 1, M)
    return root

def sumOfSubtree(root):
    if root == None:
        return 0
    sum = root.elem + sumOfSubtree(root.left) + sumOfSubtree(root.right)
    return sum


def subtract_summation(root):
    if root == None:
        return 0


    lefty_total = sumOfSubtree(root.left)
    righty_sum = sumOfSubtree(root.right)
    return lefty_total - righty_sum

def level_sum(root):
    if root == None:
        return 0
    return -root.elem - level_sum(root.left) - level_sum(root.right)

def rangeSum(root, low, high):
  #To Do
  if root == None:  
        return 0


  if root.elem < low:
        return rangeSum(root.right, low, high)


  if root.elem > high:
        return rangeSum(root.left, low, high)


  return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def inord_traverse(node):
    if node == None:
        return []


    l_part = inord_traverse(node.left)
    curr = [node.elem]
    r_part = inord_traverse(node.right)
   
    return l_part + curr + r_part




def isBST(root):
    v = inord_traverse(root)


    for x in range(1, len(v)):
