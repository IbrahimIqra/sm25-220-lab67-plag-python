def swap_child(root, level, M):
  if root == None:
    return None


  if level<= M:


    root.left, root.right = root.right, root.left


    swap_child(root.left, level+1, M)
    swap_child(root.right, level+1, M)


  return root

def subtreesum(node):
  if node == None:
    return 0
  else:
    return node.elem + subtreesum(node.left) + subtreesum(node.right)

    rightsum=subtreesum(root.right)

def level_sum(root, level=0):
    if root == None:
        return 0


    # If the level is odd


    if level % 2 == 1:
        return root.elem + level_sum(root.left, level + 1) + level_sum(root.right, level + 1)
    # If the level is even
    else:
        return -root.elem + level_sum(root.left, level + 1) + level_sum(root.right, level + 1)

def rangeSum(root, low, high):
    if root is None:
        return 0


    if root.elem < low:      
      return rangeSum(root.right, low, high)
    if root.elem > high:
      return rangeSum(root.left, low, high)
   
    return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high)

def mirror_sum(root):
  if root==None:
    return 0


  def helper(left, right):
    if not left or not right:
      return 0
    current_sum = left.elem + right.elem


    left_sum = helper(left.left, right.right)  
    right_sum = helper(left.right, right.left)
   
    total=current_sum + left_sum + right_sum
    return total
     
  return helper(root.left, root.right)

def isBST(root):
    def helper(node, min_val, max_val):
        if node is None:
            return True


        if not (min_val < node.elem < max_val):
            return False
           
        return (helper(node.left, min_val, node.elem) and
                helper(node.right, node.elem, max_val))


    return helper(root, float('-inf'), float('inf'))
