def swap_child(root, level, M):


  if root==None:
    return None
  else:


    new_root=BTNode(root.elem)
    if level<M:
      new_root.left=swap_child(root.right, level+1, M)
      new_root.right=swap_child(root.left, level+1, M)
    else:
      new_root.left=swap_child(root.left, level+1, M)
      new_root.right=swap_child(root.right, level+1, M)
  return new_root

def level_sum(root):

def level(root,current_level):
  if current_level%2==0:
    if root==None:
      return 0


    else:
      return -root.elem+level(root.left,current_level+1)+level(root.right,current_level+1)
  elif current_level%2!=0:
    if root==None:
      return 0
    else:
      return +root.elem+level(root.left,current_level+1)+level(root.right,current_level+1)

def mirror_sum(root):
  if root == None:
    return 0
  if root.left == None and root.right == None:
    return 0


  return mirror_sum_helper(root.left, root.right)




def mirror_sum_helper(left_node, right_node):
  if left_node == None or right_node == None:
    return 0
  current_sum = left_node.elem + right_node.elem


  current_sum += mirror_sum_helper(left_node.left, right_node.right)
  current_sum += mirror_sum_helper(left_node.right, right_node.left)


  return current_sum

def rangeSum(root, low, high):
  sum = 0
  def rangeSumhelper(root, low, high):
    nonlocal sum
    if root == None:
      return 0
    if low <= root.elem <= high:
      sum += root.elem
    rangeSumhelper(root.left, low, high)
    rangeSumhelper(root.right, low, high)
  rangeSumhelper(root, low, high)
  return sum

def isBST(root):
  list1 = []
  def inorder(root):
    nonlocal list1
    if root == None:
      return 0
    inorder(root.left)
    list1.append(root.elem)
    inorder(root.right)
  inorder(root)
 
  last = list1[0]
  for i in range(1,len(list1)):
    if last > list1[i]:
      return False
    last = list1[i]
  return True
